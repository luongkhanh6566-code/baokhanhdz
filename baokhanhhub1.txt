-- 🟢 Nút mở menu chính
local ScreenGui = Instance.new("ScreenGui")
local ImageButton = Instance.new("ImageButton")
local UICorner = Instance.new("UICorner")

ScreenGui.Parent = game.CoreGui
ScreenGui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling

ImageButton.Parent = ScreenGui
ImageButton.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
ImageButton.BorderSizePixel = 0
ImageButton.Position = UDim2.new(0.10615778, 0, 0.16217947, 0)
ImageButton.Size = UDim2.new(0, 40, 0, 40)
ImageButton.Draggable = true
ImageButton.Image = "https://create.roblox.com/store/asset/2377652072/Bongo-Cat-Meme"  

UICorner.CornerRadius = UDim.new(1, 10)
UICorner.Parent = ImageButton

-- 🟢 Animation tí cho nút mở menu (hiệu ứng nhấn mượt)
local TweenService = game:GetService("TweenService")
local originalSize = ImageButton.Size
ImageButton.MouseButton1Down:Connect(function()
	local pressTween = TweenService:Create(ImageButton, TweenInfo.new(0.1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Size = UDim2.new(0, 35, 0, 35)})
	pressTween:Play()
	pressTween.Completed:Connect(function()
		local releaseTween = TweenService:Create(ImageButton, TweenInfo.new(0.1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Size = originalSize})
		releaseTween:Play()
	end)
end)

ImageButton.MouseButton1Down:Connect(function()
	game:GetService("VirtualInputManager"):SendKeyEvent(true, Enum.KeyCode.End, false, game)
end)

-- 📌 SỬA PHẦN ĐÓNG/MỞ WINDOW: Xóa nút Close để chỉ ẩn (không tắt window)
task.spawn(function()
	task.wait(1.5) -- chờ GUI Fluent load xong
	for _, obj in ipairs(game.CoreGui:GetDescendants()) do
		if obj:IsA("TextButton") and (obj.Text == "✕" or obj.Text == "×" or obj.Name == "Close" or obj.Name == "Button") then
			local parentName = obj.Parent and obj.Parent.Name or ""
			if parentName:find("Title") or parentName:find("Bar") then
				obj:Destroy()
				break
			end
		end
	end
end)
